邮件发送

说明

  • 一个简单的邮件发送例子，可以直接使用。
  • 转帖请注明。

代码

package com.hexy.sendMail;

import java.util.Date;
import java.util.Properties;

import javax.mail.Address;
import javax.mail.Authenticator;
import javax.mail.Message;
import javax.mail.PasswordAuthentication;
import javax.mail.Session;
import javax.mail.Transport;
import javax.mail.internet.InternetAddress;
import javax.mail.internet.MimeMessage;

public class sendMail {
    // 发送人邮件信息
    private static String username = "zhangsan@qq.com";
    private static String password = "123456";
    // 邮件服务器IP
    private static String hostIp = "127.0.0.1";
    // 发送邮件服务器端口
    private static String hostPort = "25";
    // 发送方式
    private static String protocol = "smtp";
    //设置发件人别名
    private static String nick = "";

    public static void main(String[] args) throws Exception {
        String subject = "邮件标题";
        String body = "邮件内容";

        Properties props = System.getProperties();
        // 创建信件服务器
        props.put("mail.smtp.host", hostIp);
        props.put("mail.smtp.auth", "true");
        props.put("mail.transport.protocol", protocol);
        props.setProperty("mail.smtp.port", hostPort);

        // SSL
        props.setProperty("mail.smtp.socketFactory.class", ""); // mail.smtp.socketFactory.class
        props.setProperty("mail.smtp.socketFactory.fallback", ""); // mail.smtp.socketFactory.fallback
        props.setProperty("mail.smtp.socketFactory.port", hostPort);

        // 得到默认的对话对象
        Authenticator a = new Authenticator() {
            public PasswordAuthentication getPasswordAuthentication() {
                return new PasswordAuthentication(username, password);
            }
        };
        // 创建Session实例
        Session session = Session.getDefaultInstance(props, a);
        // 创建MimeMessage实例对象
        Message msg = new MimeMessage(session);
        // 设置发信人
        // msg.setFrom(new InternetAddress(from));
        // 设置自定义发件人昵称
        msg.setFrom(new InternetAddress(username, nick));
        // 设置收信人
        msg.setRecipients(Message.RecipientType.TO, new Address[] {
                new InternetAddress("lisi@qq.com", "收件人1"),
                new InternetAddress("wangwu@qq.com", "收件人2") });
        // 设置发送日期
        msg.setSentDate(new Date());
        // 设置邮件主题
        msg.setSubject(subject);
        // 设置邮件正文
        msg.setContent(body, "text/html;charset=gb2312");
        Transport.send(msg);
    }
}

通过SGIP协议给联通发送短信

说明

  • 该类实现了通过SGIP协议给联通发送短信的功能。相关Jar包自行准备。
  • 给联通发送短信需要去联通办理相关业务。这都是需要花钱的。而且联通的网关会将用户名密码与IP绑定，所有测试的时候需要给联通你的本机外网IP。
  • 转载须注明。

代码

package com.hexy.sendMsg;

import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.io.InputStream;
import java.io.OutputStream;
import java.net.Socket;

import spApi.Bind;
import spApi.BindResp;
import spApi.SGIP_Command;
import spApi.SGIP_Exception;
import spApi.Submit;
import spApi.SubmitResp;
import spApi.Unbind;
import spApi.UnbindResp;

public class sendUnicomMsg {

    // 联通网关IP
    private String unicomIp = "127.0.0.1";
    // 联通网关端口号
    private int unicomPort = 8808;
    // 联通登陆用户名
    private String loginName = "";
    // 联通登陆密码
    private String loginPsd = "";
    // 节点ID
    private long nodeId;
    // 企业代码
    private String cpId = "";
    // 服务代码
    private String serviceCode = "";
    // 短信接入号
    private String cpPhone = "";
    // 付费号码
    private String chargePhone = "";

    /**
     * @Description: 发送联通短信
     * @param
     * @return boolean
     */
    public boolean sendUnicomMsg() {
        int submitResult = 0;
        boolean isSend = false;
        Socket so = null;
        OutputStream out = null;
        InputStream input = null;
        Bind command = null;
        SGIP_Command sgip = null;
        SGIP_Command tmp = null;
        try {
            sgip = new SGIP_Command();
            so = new Socket(unicomIp, unicomPort);
            out = new DataOutputStream(so.getOutputStream());
            input = new DataInputStream(so.getInputStream());
            command = new Bind(nodeId,// 节点ID
                    1, // login type
                    loginName,// login name
                    loginPsd);// login password
            int loginResult = command.write(out);
            if (loginResult != 0) {
                // 登录错误
                return false;
            } else {
                // 登录成功
            }
            tmp = sgip.read(input);// 接收sgip消息
            BindResp resp = null;
            if (sgip.getCommandID() == SGIP_Command.ID_SGIP_BIND_RESP) {
                resp = (BindResp) tmp;// 强制转换为bindresp
                resp.readbody();// 对消息进行解包
                int bindResult = resp.GetResult();
                if (bindResult == 0) {
                    // 信息解包成功
                } else {
                    // 信息解包失败
                    return false;
                }
            }

            // 开始发送短信
            submitResult = submit("15611111111", "短信内容", out, input, sgip);

            if (submitResult == 0) {
                // 发送成功
                isSend = true;
            } else {
                // 发送失败
            }

            Unbind term = null;
            UnbindResp Unresp = null;
            term = new Unbind(nodeId);
            term.write(out);// 给联通发送退出请求
            tmp = sgip.read(input);
            if (sgip.getCommandID() == SGIP_Command.ID_SGIP_UNBIND_RESP) {
                Unresp = (UnbindResp) tmp;
                int unbindFlag = Unresp.GetFlag();
                // 退出的标志
                System.out.println(unbindFlag);
            }
            input.close();
            out.close();
            so.close();
        } catch (SGIP_Exception e) {
            System.out.println("短信发送失败，SGIP_Exception异常：" + e);
        } catch (Exception e) {
            System.out.println("短信发送失败 ，Exception： " + e);
        } finally {
            try {
                if (input != null) {
                    input.close();
                }
                if (out != null) {
                    out.close();
                }
                if (so != null) {
                    so.close();
                }
            } catch (IOException e) {
                System.out
                        .println("Finally IOException, exception info : " + e);
            }
        }
        return isSend;
    }

    /**
     * @Description: 提交短信至联通网关
     * @param
     * @return boolean
     */
    public int submit(String phoneNum, String msg, OutputStream out,
            InputStream input, SGIP_Command sgip) {
        int submitResult = -1;
        SGIP_Command tmp = null;
        Submit submit = null;
        SubmitResp submitresp = null;
        try {
            submit = new Submit(nodeId,// node id同上
                    cpPhone,// cp_phone
                    chargePhone,// 付费号码
                    1,// 接收短消息的手机数
                    phoneNum,//
                    cpId,// cp_id
                    serviceCode,// 业务代码
                    0,// 计费类型
                    "1",// 短消息收费值
                    "0",// 赠送话费
                    1,// 代收标志
                    8,// 引起MT的原因
                    9,// 优先级
                    "",// 短消息终止时间
                    "",// 011125120000032+短消息定时发送时间
                    1,// 状态报告标志
                    0,// GSM协议类型
                    0,// GSM协议类型
                    15,// 短消息编码格式 GBK
                    0,// 信息类型
                    msg.length(),// 短消息长度
                    msg);// 短消息内容
            submit.write(out);// 发送submit
            tmp = sgip.read(input);
            if (tmp.getCommandID() == SGIP_Command.ID_SGIP_SUBMIT_RESP) {
                submitresp = (SubmitResp) tmp;// 强制转换
                submitresp.readbody();// 解包
                int seqNum = submitresp.getSeqno_2();
                submitResult = submitresp.getResult();
            }
        } catch (SGIP_Exception e) {
            System.out.println("SGIP_Exception, exception info = " + e);
        } catch (IOException e) {
            System.out.println("IOException, exception info = " + e);
        } catch (Exception e) {
            System.out.println("Exception, exception info = " + e);
        }
        return submitResult;
    }

}

解决Mybatis获取Blob类型字段乱码问题

Mybatis 在获取Blob类型的字段时会出现乱码。这里需要用到Mybatis的拦截器进行数据类型转换。
首先需要知道的是BaseTypeHandler是Mybatis提供的基础转换类，该类实现了TypeHandler接口并且提供了很多实用的方法。因此，建议每个子一定转换类都继承BaseTypeHandler。
废话不说，直接上代码：

package com.he.webx.utils;

import java.io.ByteArrayInputStream;
import java.io.UnsupportedEncodingException;
import java.sql.Blob;
import java.sql.CallableStatement;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import org.apache.ibatis.type.BaseTypeHandler;
import org.apache.ibatis.type.JdbcType;

/**
 * 自定义typehandler，解决mybatis存储blob字段后，出现乱码的问题 
 * mapper.xml中result配置写法如下 <result typeHandler="com.he.webx.utils.ConvertBlobTypeHandler"/>
 */
public class ConvertBlobTypeHandler extends BaseTypeHandler<String> {
// ###指定字符集
private static final String DEFAULT_CHARSET = "utf-8";

@Override
public void setNonNullParameter(PreparedStatement ps, int i,
        String parameter, JdbcType jdbcType) throws SQLException {
    ByteArrayInputStream bis;
    try {
        // ###把String转化成byte流
        bis = new ByteArrayInputStream(parameter.getBytes(DEFAULT_CHARSET));
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException("Blob Encoding Error!");
    }
    ps.setBinaryStream(i, bis, parameter.length());
}

@Override
public String getNullableResult(ResultSet rs, String columnName)
        throws SQLException {
    Blob blob = rs.getBlob(columnName);
    byte[] returnValue = null;
    if (null != blob) {
        returnValue = blob.getBytes(1, (int) blob.length());
    }
    try {
        // ###把byte转化成string
        return new String(returnValue, DEFAULT_CHARSET);
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException("Blob Encoding Error!");
    }
}

@Override
public String getNullableResult(CallableStatement cs, int columnIndex)
        throws SQLException {
    Blob blob = cs.getBlob(columnIndex);
    byte[] returnValue = null;
    if (null != blob) {
        returnValue = blob.getBytes(1, (int) blob.length());
    }
    try {
        return new String(returnValue, DEFAULT_CHARSET);
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException("Blob Encoding Error!");
    }
}

@Override
public String getNullableResult(ResultSet arg0, int arg1)
        throws SQLException {
    // TODO Auto-generated method stub
    return null;
}
}

本地环境为Mysql,字段类型为Blob,LongBlob,均可以使用。
在Mybatis的Mapper.xml文件中的resultMap的result中写法为 : 

<result property="XX" column="XXX" typeHandler="com.he.webx.utils.ConvertBlobTypeHandler" />

setTimeout和setInterval

这2个函数用法和功能都很简单，无非就是setTimeout(code,millisec)，这里需要注意的就是code，无论是function还是函数名，都需要用"",括起来，否则是不起作用的。如下 

<script> 
    $(function(){  
        setTimeout("function(){  
            alert(1);  
        }",1000);
        setInterval("function(){  
            alert(2);
        }",2000);
    })
</script>

需要将function用""括起来。这是因为这2个函数的第一个参数接受的是字符串，最终是通过类似于eval机制执行。因此，setTimeout和setInterval的第一个参数需要传入的是字符串。
